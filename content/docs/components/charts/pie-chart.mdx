## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/pie-chart.json
```

## Usage

```tsx
import { PieChart } from "@/components/ui/pie-chart";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <PieChart {...demoProps} />;
}
```
