## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/gauge.json
```

## Usage

```tsx
import { Gauge } from "@/components/ui/gauge";

export function Example() {
  return <Gauge value={72} max={100} label="CPU Usage" size="md" />;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop    | Type                   | Default     | Description                          |
| ------- | ---------------------- | ----------- | ------------------------------------ |
| `value` | `number`               | `required`  | Current value to display             |
| `min`   | `number`               | `0`         | Minimum value of the gauge range     |
| `max`   | `number`               | `100`       | Maximum value of the gauge range     |
| `label` | `string`               | `undefined` | Text label displayed below the gauge |
| `color` | `string`               | `undefined` | Custom color for the filled portion  |
| `size`  | `"sm" \| "md" \| "lg"` | `"md"`      | Gauge size: small, medium, or large  |

## Examples

### Small Inline Gauge

```tsx
<Gauge value={45} size="sm" label="Memory" />
```

### Large Gauge with Custom Range

```tsx
<Gauge value={3200} min={0} max={5000} size="lg" label="RPM" color="#d97706" />
```

### Colored Warning Gauge

```tsx
<Gauge value={92} max={100} size="md" label="Disk Usage" color="#dc2626" />
```
