## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/sparkline.json
```

## Usage

```tsx
import { Sparkline } from "@/components/ui/sparkline";

export function Example() {
  return (
    <Sparkline
      data={[4, 8, 15, 16, 23, 42, 38, 26, 19, 12]}
      width={20}
      label="Requests"
    />
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop    | Type       | Default     | Description                           |
| ------- | ---------- | ----------- | ------------------------------------- |
| `data`  | `number[]` | `required`  | Array of numeric values to plot       |
| `width` | `number`   | `20`        | Maximum width in characters           |
| `color` | `string`   | `undefined` | Custom color for the sparkline        |
| `label` | `string`   | `undefined` | Text label displayed before the chart |

## Examples

### Minimal Sparkline

```tsx
<Sparkline data={[1, 3, 2, 5, 4, 6, 3]} />
```

### Colored with Label

```tsx
<Sparkline
  data={[20, 40, 35, 50, 45, 60, 55, 70]}
  label="CPU"
  color="#dc2626"
  width={16}
/>
```

### Wide Sparkline

```tsx
<Sparkline
  data={[10, 12, 8, 14, 9, 16, 11, 18, 13, 20, 15, 22]}
  width={40}
  label="Throughput"
  color="#16a34a"
/>
```
