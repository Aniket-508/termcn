## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/line-chart.json
```

## Usage

```tsx
import { LineChart } from "@/components/ui/line-chart";

export function Example() {
  return (
    <LineChart
      data={[
        { label: "Jan", value: 30 },
        { label: "Feb", value: 45 },
        { label: "Mar", value: 28 },
        { label: "Apr", value: 60 },
        { label: "May", value: 52 },
        { label: "Jun", value: 75 },
      ]}
      width={40}
      height={10}
      title="Monthly Sales"
    />
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type                   | Default     | Description                                    |
| ---------- | ---------------------- | ----------- | ---------------------------------------------- |
| `data`     | `LineChartDataPoint[]` | `required`  | Array of numbers or `{ label, value }` objects |
| `width`    | `number`               | `40`        | Total chart width in columns                   |
| `height`   | `number`               | `10`        | Chart height in rows                           |
| `title`    | `string`               | `undefined` | Optional title displayed above chart           |
| `color`    | `string`               | `undefined` | Custom color for the plot line                 |
| `showAxes` | `boolean`              | `true`      | Show X and Y axes with labels                  |

## Examples

### Simple Numeric Data

```tsx
<LineChart data={[3, 7, 4, 8, 2, 9, 5]} width={30} height={8} />
```

### Without Axes

```tsx
<LineChart
  data={[10, 25, 18, 42, 33, 55, 48]}
  showAxes={false}
  width={36}
  height={6}
  color="#16a34a"
/>
```

### Labeled Data Points

```tsx
<LineChart
  data={[
    { label: "W1", value: 100 },
    { label: "W2", value: 140 },
    { label: "W3", value: 120 },
    { label: "W4", value: 180 },
  ]}
  width={32}
  height={10}
  title="Weekly Users"
/>
```
