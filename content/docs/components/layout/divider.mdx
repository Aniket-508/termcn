## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/divider.json
```

## Usage

```tsx
import { Divider } from "@/components/ui/divider";

export function Example() {
  return (
    <Box flexDirection="column">
      <Text>Above</Text>
      <Divider label="Section" />
      <Text>Below</Text>
    </Box>
  );
}
```

### npm Dependencies

## Props

| Prop          | Type                             | Default        | Description                                               |
| ------------- | -------------------------------- | -------------- | --------------------------------------------------------- |
| `variant`     | `"single" \| "double" \| "bold"` | `"single"`     | Line character style (─, ═, or ━)                         |
| `orientation` | `"horizontal" \| "vertical"`     | `"horizontal"` | Direction the divider draws                               |
| `color`       | `string`                         | `undefined`    | Custom color override; defaults to the theme border color |
| `label`       | `string`                         | `undefined`    | Optional inline text label for horizontal dividers        |
| `height`      | `number`                         | `1`            | Height in rows for vertical dividers                      |
| `width`       | `number`                         | `undefined`    | Width in columns for horizontal dividers                  |

## Examples

### Simple horizontal divider

```tsx
<Divider />
```

### Bold divider with label

```tsx
<Divider variant="bold" label="Settings" />
```

### Double-line divider with fixed width

```tsx
<Divider variant="double" width={40} />
```

### Vertical divider

```tsx
<Box flexDirection="row">
  <Text>Left</Text>
  <Divider orientation="vertical" height={3} />
  <Text>Right</Text>
</Box>
```
