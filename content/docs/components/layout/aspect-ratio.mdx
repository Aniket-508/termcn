## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/aspect-ratio.json
```

## Usage

```tsx
import { AspectRatio } from "@/components/ui/aspect-ratio";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <AspectRatio {...demoProps} />;
}
```

## Props

| Prop       | Type        | Default    | Description |
| ---------- | ----------- | ---------- | ----------- |
| `children` | `ReactNode` | `required` | -           |
| `ratio`    | `number`    | `16 / 9`   | -           |
| `width`    | `number`    | `80`       | -           |
