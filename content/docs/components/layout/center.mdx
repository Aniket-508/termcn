## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/center.json
```

## Usage

```tsx
import { Center } from "@/components/ui/center";

export function Example() {
  return (
    <Center>
      <Text>Centered content</Text>
    </Center>
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type                                   | Default  | Description                             |
| ---------- | -------------------------------------- | -------- | --------------------------------------- |
| `children` | `ReactNode`                            | required | Content to center within the container  |
| `axis`     | `"both" \| "horizontal" \| "vertical"` | `"both"` | Axis along which to center the children |

## Examples

### Center on both axes (default)

```tsx
<Center>
  <Text>Centered everywhere</Text>
</Center>
```

### Horizontal centering only

```tsx
<Center axis="horizontal">
  <Text>Horizontally centered</Text>
</Center>
```

### Vertical centering only

```tsx
<Center axis="vertical">
  <Text>Vertically centered</Text>
</Center>
```
