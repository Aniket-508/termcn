## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/app-shell.json
```

## Usage

```tsx
import { AppShell } from "@/components/ui/app-shell";

export function Example() {
  return (
    <AppShell>
      <AppShell.Header>
        <AppShell.Tip>Press Tab to autocomplete commands</AppShell.Tip>
      </AppShell.Header>
      <AppShell.Input
        placeholder="Enter a command..."
        prefix="$"
        onSubmit={(value) => console.log(value)}
      />
      <AppShell.Content height={16}>
        {/* scrollable output area */}
      </AppShell.Content>
      <AppShell.Hints items={["↑↓ scroll", "enter submit", "esc quit"]} />
    </AppShell>
  );
}
```

## Props

### AppShell

| Prop         | Type        | Default     | Description                                   |
| ------------ | ----------- | ----------- | --------------------------------------------- |
| `children`   | `ReactNode` | `required`  | Layout content composed from sub-components.  |
| `fullscreen` | `boolean`   | `undefined` | Expand the shell to fill the entire terminal. |

### AppShell.Header

| Prop       | Type        | Default    | Description                               |
| ---------- | ----------- | ---------- | ----------------------------------------- |
| `children` | `ReactNode` | `required` | Header content, typically contains `Tip`. |

### AppShell.Tip

| Prop       | Type        | Default    | Description                          |
| ---------- | ----------- | ---------- | ------------------------------------ |
| `children` | `ReactNode` | `required` | Dimmed tip text shown in the header. |

### AppShell.Input

| Prop          | Type                                        | Default               | Description                                      |
| ------------- | ------------------------------------------- | --------------------- | ------------------------------------------------ |
| `value`       | `string`                                    | `undefined`           | Controlled input value.                          |
| `onChange`    | `(value: string) => void`                   | `undefined`           | Callback fired on each keystroke.                |
| `onSubmit`    | `(value: string) => void`                   | `undefined`           | Callback fired when Enter is pressed.            |
| `placeholder` | `string`                                    | `"Type something..."` | Dimmed placeholder shown when input is empty.    |
| `borderStyle` | `"single" \| "double" \| "round" \| "bold"` | `"single"`            | Border style around the input box.               |
| `borderColor` | `string`                                    | `undefined`           | Border color override; defaults to theme border. |
| `prefix`      | `string`                                    | `">"`                 | Prompt character displayed before the input.     |

### AppShell.Content

| Prop         | Type        | Default     | Description                                        |
| ------------ | ----------- | ----------- | -------------------------------------------------- |
| `children`   | `ReactNode` | `required`  | Scrollable body content.                           |
| `autoscroll` | `boolean`   | `undefined` | Automatically scroll to the bottom on new content. |
| `height`     | `number`    | `20`        | Visible height in rows.                            |

### AppShell.Hints

| Prop       | Type        | Default     | Description                                         |
| ---------- | ----------- | ----------- | --------------------------------------------------- |
| `items`    | `string[]`  | `undefined` | Shortcut hints joined with `\|` separators.         |
| `children` | `ReactNode` | `undefined` | Custom hint content (used when `items` is not set). |

## Examples

### Minimal shell with input

```tsx
<AppShell>
  <AppShell.Input placeholder="Search..." prefix="?" />
</AppShell>
```

### With header tip and hints

```tsx
<AppShell>
  <AppShell.Header>
    <AppShell.Tip>Use arrow keys to navigate history</AppShell.Tip>
  </AppShell.Header>
  <AppShell.Input onSubmit={(v) => run(v)} />
  <AppShell.Hints items={["↑↓ history", "tab complete", "ctrl+c exit"]} />
</AppShell>
```

### Custom border style

```tsx
<AppShell>
  <AppShell.Input
    borderStyle="round"
    borderColor="cyan"
    prefix="λ"
    placeholder="Type an expression..."
  />
  <AppShell.Content height={10}>
    {output.map((line, i) => (
      <Text key={i}>{line}</Text>
    ))}
  </AppShell.Content>
</AppShell>
```
