## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/login-flow.json
```

## Usage

```tsx
import { LoginFlow } from "@/components/ui/login-flow";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <LoginFlow {...demoProps} />;
}
```
