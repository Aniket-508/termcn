## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/splash-screen.json
```

## Usage

```tsx
import { SplashScreen } from "@/components/ui/splash-screen";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <SplashScreen {...demoProps} />;
}
```

## Props

| Prop            | Type                                        | Default     | Description |
| --------------- | ------------------------------------------- | ----------- | ----------- |
| `title`         | `string`                                    | `required`  | -           |
| `font`          | `FigletFont`                                | `"block"`   | -           |
| `titleColor`    | `string`                                    | `undefined` | -           |
| `titleColorAlt` | `string`                                    | `undefined` | -           |
| `bold`          | `boolean`                                   | `undefined` | -           |
| `subtitle`      | `string`                                    | `undefined` | -           |
| `subtitleDim`   | `boolean`                                   | `true`      | -           |
| `author`        | `&#123; name: string; href?: string &#125;` | `undefined` | -           |
| `statusLine`    | `ReactNode`                                 | `undefined` | -           |
| `padding`       | `number`                                    | `2`         | -           |
| `align`         | `"left" \| "center"`                        | `undefined` | -           |
