## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/setup-flow.json
```

## Usage

```tsx
import { SetupFlow } from "@/components/ui/setup-flow";

export function Example() {
  return (
    <SetupFlow title="create-app">
      <SetupFlow.Badge label="v1.0.0" />
      <SetupFlow.Step status="done">Project name set to my-app</SetupFlow.Step>
      <SetupFlow.Step status="done">Template selected: React</SetupFlow.Step>
      <SetupFlow.Step status="active">
        Installing dependencies...
      </SetupFlow.Step>
      <SetupFlow.MultiSelect
        label="Select features"
        hint="space to toggle, enter to confirm"
        options={[
          { value: "ts", label: "TypeScript" },
          { value: "eslint", label: "ESLint", description: "Code linting" },
          { value: "prettier", label: "Prettier", description: "Formatting" },
        ]}
        onSubmit={(values) => console.log(values)}
      />
    </SetupFlow>
  );
}
```

## Props

### SetupFlow

| Prop             | Type          | Default     | Description                                           |
| ---------------- | ------------- | ----------- | ----------------------------------------------------- |
| `title`          | `string`      | `undefined` | ASCII art title rendered at the top of the flow.      |
| `titleFont`      | `BigTextFont` | `"block"`   | Font variant for the title.                           |
| `titleColor`     | `string`      | `"#888888"` | Primary color for the title text.                     |
| `titleColorAlt`  | `string`      | `undefined` | Alternate row color for a striped title effect.       |
| `connectorChar`  | `string`      | `"│"`       | Character drawn between steps.                        |
| `connectorColor` | `string`      | `undefined` | Color of the connector; defaults to muted foreground. |
| `children`       | `ReactNode`   | `required`  | Step, Badge, Spinner, and MultiSelect sub-components. |

### SetupFlow.Badge

| Prop    | Type     | Default    | Description                           |
| ------- | -------- | ---------- | ------------------------------------- |
| `label` | `string` | `required` | Text displayed inside the badge pill. |
| `bg`    | `string` | `"cyan"`   | Background color of the badge.        |
| `color` | `string` | `"black"`  | Foreground text color of the badge.   |

### SetupFlow.Step

| Prop        | Type                                                      | Default     | Description                                         |
| ----------- | --------------------------------------------------------- | ----------- | --------------------------------------------------- |
| `icon`      | `string`                                                  | `undefined` | Custom icon character; overrides the status icon.   |
| `iconColor` | `string`                                                  | `undefined` | Color for the icon; overrides status-derived color. |
| `status`    | `"done" \| "active" \| "pending" \| "success" \| "error"` | `"done"`    | Visual state determining the icon and dimming.      |
| `children`  | `ReactNode`                                               | `required`  | Step label text.                                    |

### SetupFlow.Spinner

| Prop         | Type         | Default     | Description                                |
| ------------ | ------------ | ----------- | ------------------------------------------ |
| `label`      | `string`     | `required`  | Text shown next to the animated spinner.   |
| `onComplete` | `() => void` | `undefined` | Callback fired when the spinner completes. |

### SetupFlow.MultiSelect

| Prop            | Type                                                       | Default     | Description                                       |
| --------------- | ---------------------------------------------------------- | ----------- | ------------------------------------------------- |
| `label`         | `string`                                                   | `required`  | Prompt text above the options.                    |
| `hint`          | `string`                                                   | `undefined` | Helper text shown in parentheses after the label. |
| `options`       | `{ value: string; label: string; description?: string }[]` | `required`  | Selectable option definitions.                    |
| `values`        | `string[]`                                                 | `undefined` | Controlled selected values.                       |
| `onChange`      | `(values: string[]) => void`                               | `undefined` | Callback fired when selection changes.            |
| `onSubmit`      | `(values: string[]) => void`                               | `undefined` | Callback fired when Enter confirms the selection. |
| `checkedChar`   | `string`                                                   | `"■"`       | Character for checked options.                    |
| `uncheckedChar` | `string`                                                   | `"□"`       | Character for unchecked options.                  |
| `checkedColor`  | `string`                                                   | `"green"`   | Color for the checked indicator.                  |

## Examples

### Completed steps

```tsx
<SetupFlow>
  <SetupFlow.Step status="success">Dependencies installed</SetupFlow.Step>
  <SetupFlow.Step status="success">Config files created</SetupFlow.Step>
  <SetupFlow.Step status="done">Ready to go!</SetupFlow.Step>
</SetupFlow>
```

### Active spinner step

```tsx
<SetupFlow title="init" titleColor="cyan">
  <SetupFlow.Badge label="create-app" bg="green" />
  <SetupFlow.Step status="done">Cloned template</SetupFlow.Step>
  <SetupFlow.Spinner label="Installing packages..." />
</SetupFlow>
```

### Error state

```tsx
<SetupFlow>
  <SetupFlow.Step status="done">Checked permissions</SetupFlow.Step>
  <SetupFlow.Step status="error">Network connection failed</SetupFlow.Step>
  <SetupFlow.Step status="pending">Deploy to server</SetupFlow.Step>
</SetupFlow>
```
