## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/help-screen.json
```

## Usage

```tsx
import { HelpScreen } from "@/components/ui/help-screen";

export function Example() {
  return (
    <HelpScreen
      title="mycli"
      tagline="A blazing-fast task runner"
      usage="mycli <command> [options]"
    >
      <HelpScreen.Section label="Commands">
        <HelpScreen.Row flag="init" description="Create a new project" />
        <HelpScreen.Row flag="build" description="Compile the source files" />
        <HelpScreen.Row flag="test" description="Run the test suite" />
      </HelpScreen.Section>
      <HelpScreen.Section label="Options">
        <HelpScreen.Row
          flag="--help, -h"
          description="Show this help message"
        />
        <HelpScreen.Row
          flag="--version, -v"
          description="Print version number"
        />
        <HelpScreen.Row flag="--verbose" description="Enable verbose logging" />
      </HelpScreen.Section>
    </HelpScreen>
  );
}
```

## Props

### HelpScreen

| Prop          | Type          | Default     | Description                                              |
| ------------- | ------------- | ----------- | -------------------------------------------------------- |
| `title`       | `string`      | `required`  | CLI name rendered as a large ASCII art banner.           |
| `font`        | `BigTextFont` | `"block"`   | Font used for the title banner.                          |
| `titleColor`  | `string`      | `undefined` | Color of the title text; defaults to theme primary.      |
| `tagline`     | `string`      | `undefined` | Short description shown below the title.                 |
| `usage`       | `string`      | `undefined` | Usage pattern string, e.g. `"mycli <cmd> [opts]"`.       |
| `description` | `string`      | `undefined` | Longer description paragraph below the usage line.       |
| `columnGap`   | `number`      | `4`         | Horizontal gap between the flag and description columns. |
| `flagWidth`   | `number`      | `undefined` | Fixed flag column width; auto-computed when omitted.     |
| `children`    | `ReactNode`   | `required`  | One or more `HelpScreen.Section` elements.               |

### HelpScreen.Section

| Prop         | Type        | Default     | Description                                        |
| ------------ | ----------- | ----------- | -------------------------------------------------- |
| `label`      | `string`    | `required`  | Section heading text (e.g. "Commands", "Options"). |
| `labelColor` | `string`    | `undefined` | Color override for the section heading.            |
| `children`   | `ReactNode` | `required`  | One or more `HelpScreen.Row` elements.             |

### HelpScreen.Row

| Prop               | Type     | Default     | Description                                    |
| ------------------ | -------- | ----------- | ---------------------------------------------- |
| `flag`             | `string` | `required`  | Flag or command name shown in the left column. |
| `description`      | `string` | `required`  | Explanation text in the right column.          |
| `flagColor`        | `string` | `undefined` | Color for the flag text; defaults to muted.    |
| `descriptionColor` | `string` | `undefined` | Color for the description text.                |

## Examples

### Minimal help

```tsx
<HelpScreen title="serve" usage="serve [dir] [--port N]">
  <HelpScreen.Section label="Options">
    <HelpScreen.Row flag="--port, -p" description="Port to listen on" />
    <HelpScreen.Row flag="--host" description="Hostname to bind" />
  </HelpScreen.Section>
</HelpScreen>
```

### With tagline and description

```tsx
<HelpScreen
  title="deploy"
  tagline="Ship your app to the cloud"
  usage="deploy <target>"
  description="Deploys the current project to the specified target environment."
>
  <HelpScreen.Section label="Targets">
    <HelpScreen.Row flag="production" description="Live environment" />
    <HelpScreen.Row flag="staging" description="Pre-release testing" />
  </HelpScreen.Section>
</HelpScreen>
```

### Custom colors

```tsx
<HelpScreen title="lint" titleColor="yellow" font="simple">
  <HelpScreen.Section label="Rules" labelColor="green">
    <HelpScreen.Row
      flag="--fix"
      description="Auto-fix violations"
      flagColor="cyan"
    />
    <HelpScreen.Row
      flag="--strict"
      description="Enable strict mode"
      flagColor="cyan"
    />
  </HelpScreen.Section>
</HelpScreen>
```
