## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/popover.json
```

## Usage

```tsx
import { Popover } from "@/components/ui/popover";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Popover {...demoProps} />;
}
```
