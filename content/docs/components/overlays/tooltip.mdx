## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tooltip.json
```

## Usage

```tsx
import { Tooltip } from "@/components/ui/tooltip";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Tooltip {...demoProps} />;
}
```

## Props

| Prop          | Type                                                                                            | Default     | Description                                                          |
| ------------- | ----------------------------------------------------------------------------------------------- | ----------- | -------------------------------------------------------------------- |
| `children`    | `ReactNode`                                                                                     | `required`  | -                                                                    |
| `content`     | `string`                                                                                        | `required`  | -                                                                    |
| `position`    | `"top" \| "bottom" \| "left" \| "right"`                                                        | `"top"`     | -                                                                    |
| `isVisible`   | `boolean`                                                                                       | `false`     | -                                                                    |
| `borderStyle` | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"single"`  | Border style of the tooltip box. Default: 'single'                   |
| `borderColor` | `string`                                                                                        | `undefined` | Border/arrow color. Default: theme.colors.border                     |
| `paddingX`    | `number`                                                                                        | `1`         | Horizontal padding of the tooltip box. Default: 1                    |
| `paddingY`    | `number`                                                                                        | `0`         | Vertical padding of the tooltip box. Default: 0                      |
| `gap`         | `number`                                                                                        | `1`         | Gap between tooltip and trigger for left/right positions. Default: 1 |
| `arrowDown`   | `string`                                                                                        | `"↓"`       | Arrow shown below tooltip (top position). Default: '↓'               |
| `arrowUp`     | `string`                                                                                        | `"↑"`       | Arrow shown above tooltip (bottom position). Default: '↑'            |
