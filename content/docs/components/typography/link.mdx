## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/link.json
```

## Usage

```tsx
import { Link } from "@/components/ui/link";

export function Example() {
  return <Link href="https://github.com">GitHub</Link>;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type        | Default     | Description                                |
| ---------- | ----------- | ----------- | ------------------------------------------ |
| `children` | `ReactNode` | `required`  | Display text for the link                  |
| `href`     | `string`    | `required`  | URL the link points to                     |
| `color`    | `string`    | `undefined` | Color override for the link text           |
| `showHref` | `boolean`   | `false`     | Show the URL in parentheses after the text |

## Examples

### Basic Usage

```tsx
<Link href="https://example.com">Visit Example</Link>
```

### With Visible URL

```tsx
<Link href="https://docs.example.com/api" showHref>
  API Docs
</Link>
```

### Custom Color

```tsx
<Link href="https://github.com" color="#e74c3c">
  GitHub
</Link>
```
