## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/code.json
```

## Usage

```tsx
import { Code } from "@/components/ui/code";

export function Example() {
  return (
    <Code language="typescript">
      {`const greeting = "Hello, world!";\nconsole.log(greeting);`}
    </Code>
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop                  | Type                                                                                            | Default     | Description                                         |
| --------------------- | ----------------------------------------------------------------------------------------------- | ----------- | --------------------------------------------------- |
| `children`            | `string`                                                                                        | `required`  | Source code string to display                       |
| `language`            | `string`                                                                                        | `undefined` | Language label shown in the top-right corner        |
| `inline`              | `boolean`                                                                                       | `false`     | Render as an inline code snippet instead of a block |
| `borderStyle`         | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"single"`  | Border style for the code block frame               |
| `showLineNumbers`     | `boolean`                                                                                       | `true`      | Whether to show line numbers in the gutter          |
| `lineNumberSeparator` | `string`                                                                                        | `"│ "`      | Separator string between line numbers and code      |
| `keywordColor`        | `string`                                                                                        | `undefined` | Override color for keyword tokens                   |
| `stringColor`         | `string`                                                                                        | `undefined` | Override color for string literal tokens            |
| `numberColor`         | `string`                                                                                        | `undefined` | Override color for number tokens                    |
| `commentColor`        | `string`                                                                                        | `undefined` | Override color for comment tokens                   |
| `operatorColor`       | `string`                                                                                        | `undefined` | Override color for operator tokens                  |
| `plainColor`          | `string`                                                                                        | `undefined` | Override color for plain/identifier tokens          |

## Examples

### Basic Usage

```tsx
<Code language="javascript">{`const x = 42;`}</Code>
```

### Inline Code

```tsx
<Code inline>{`npm install ink`}</Code>
```

### Without Line Numbers

```tsx
<Code language="python" showLineNumbers={false}>
  {`def greet(name):\n    return f"Hello, {name}"`}
</Code>
```

### Custom Token Colors

```tsx
<Code
  language="typescript"
  keywordColor="#c678dd"
  stringColor="#98c379"
  numberColor="#d19a66"
>
  {`const count = 10;\nconst msg = "done";`}
</Code>
```
