## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/digits.json
```

## Usage

```tsx
import { Digits } from "@/components/ui/digits";

export function Example() {
  return <Digits value="12:30" />;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop    | Type               | Default     | Description                                        |
| ------- | ------------------ | ----------- | -------------------------------------------------- |
| `value` | `string \| number` | `required`  | Numeric value or string to render as large digits  |
| `color` | `string`           | `undefined` | Color override for the digit characters            |
| `size`  | `DigitSize`        | `"md"`      | Digit size: `"sm"` (plain text), `"md"`, or `"lg"` |

## Examples

### Basic Usage

```tsx
<Digits value={42} />
```

### Large Digits

```tsx
<Digits value="09:45" size="lg" />
```

### Small Inline Digits

```tsx
<Digits value={100} size="sm" />
```

### With Custom Color

```tsx
<Digits value="3.14" color="#61afef" />
```
