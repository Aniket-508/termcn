## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/big-text.json
```

## Usage

```tsx
import { BigText } from "@/components/ui/big-text";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <BigText {...demoProps} />;
}
```
