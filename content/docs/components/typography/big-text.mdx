## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/big-text.json
```

## Usage

```tsx
import { BigText } from "@/components/ui/big-text";

export function Example() {
  return <BigText>HELLO</BigText>;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type          | Default     | Description                                          |
| ---------- | ------------- | ----------- | ---------------------------------------------------- |
| `children` | `string`      | `required`  | Text string to render as large ASCII art             |
| `color`    | `string`      | `undefined` | Color override for the rendered characters           |
| `font`     | `BigTextFont` | `"block"`   | Font style: `"block"` (solid) or `"simple"` (shaded) |

## Examples

### Basic Usage

```tsx
<BigText>HELLO</BigText>
```

### With Custom Color

```tsx
<BigText color="#ff6b6b">ERROR</BigText>
```

### Simple Font

```tsx
<BigText font="simple">DONE</BigText>
```
