## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/badge.json
```

## Usage

```tsx
import { Badge } from "@/components/ui/badge";

export function Example() {
  <Badge variant="success">Active</Badge>;
}
```

## Dependencies

### npm Dependencies

## Props

| Prop          | Type                                                                                            | Default     | Description                                   |
| ------------- | ----------------------------------------------------------------------------------------------- | ----------- | --------------------------------------------- |
| `children`    | `string`                                                                                        | `required`  | -                                             |
| `variant`     | `BadgeVariant`                                                                                  | `"default"` | -                                             |
| `color`       | `string`                                                                                        | `undefined` | -                                             |
| `bold`        | `boolean`                                                                                       | `false`     | -                                             |
| `bordered`    | `boolean`                                                                                       | `true`      | Show a border around the badge. Default: true |
| `borderStyle` | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"round"`   | Border style. Default: 'round'                |
| `paddingX`    | `number`                                                                                        | `1`         | Horizontal padding. Default: 1                |
