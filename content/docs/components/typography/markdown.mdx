## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/markdown.json
```

## Usage

```tsx
import { Markdown } from "@/components/ui/markdown";

export function Example() {
  return (
    <Markdown>{`# Welcome\n\nThis is **bold** and *italic* text.`}</Markdown>
  );
}
```

## Dependencies

### npm Dependencies

- `ink`
- `marked`

## Props

| Prop       | Type     | Default     | Description                         |
| ---------- | -------- | ----------- | ----------------------------------- |
| `children` | `string` | `required`  | Markdown string to parse and render |
| `width`    | `number` | `undefined` | Max width for text wrapping         |

## Examples

### Basic Usage

```tsx
<Markdown>
  {`# Hello World\n\nA paragraph with **bold** and *italic* text.`}
</Markdown>
```

### Lists and Blockquotes

```tsx
<Markdown>
  {`> Important note\n\n- First item\n- Second item\n- Third item`}
</Markdown>
```

### With Width Constraint

```tsx
<Markdown width={60}>
  {`## Heading\n\nThis long paragraph will wrap at 60 characters to keep output readable in narrow terminals.`}
</Markdown>
```
