## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/markdown.json
```

## Usage

```tsx
import { Markdown } from "@/components/ui/markdown";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Markdown {...demoProps} />;
}
```
