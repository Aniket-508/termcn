## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tag.json
```

## Usage

```tsx
import { Tag } from "@/components/ui/tag";

export function Example() {
  return <Tag>typescript</Tag>;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type         | Default     | Description                                |
| ---------- | ------------ | ----------- | ------------------------------------------ |
| `children` | `ReactNode`  | `required`  | Content displayed inside the tag           |
| `onRemove` | `() => void` | `undefined` | Callback when remove button is clicked     |
| `color`    | `string`     | `undefined` | Color override for the tag text and border |
| `variant`  | `TagVariant` | `"default"` | Visual style: `"default"` or `"outline"`   |

## Examples

### Basic Usage

```tsx
<Tag>react</Tag>
```

### Outline Variant

```tsx
<Tag variant="outline">draft</Tag>
```

### With Remove Handler

```tsx
<Tag onRemove={() => console.log("removed")}>removable</Tag>
```

### Custom Color

```tsx
<Tag color="#e74c3c">urgent</Tag>
```
