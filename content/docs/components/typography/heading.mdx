## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/heading.json
```

## Usage

```tsx
import { Heading } from "@/components/ui/heading";

export function Example() {
  return <Heading level={1}>Getting Started</Heading>;
}
```

### npm Dependencies

## Props

| Prop        | Type           | Default     | Description                               |
| ----------- | -------------- | ----------- | ----------------------------------------- |
| `level`     | `HeadingLevel` | `1`         | Heading level: `1`, `2`, `3`, or `4`      |
| `children`  | `ReactNode`    | `required`  | Content to display as the heading text    |
| `color`     | `string`       | `undefined` | Color override for the heading text       |
| `prefix1`   | `string`       | `"‚Ėą‚Ėą "`     | Prefix symbol for level 1 headings        |
| `prefix2`   | `string`       | `"‚ĖĆ "`      | Prefix symbol for level 2 headings        |
| `prefix3`   | `string`       | `"‚Äļ "`      | Prefix symbol for level 3 headings        |
| `uppercase` | `boolean`      | `true`      | Whether to uppercase level 1 heading text |

## Examples

### Level 1 (Default)

```tsx
<Heading>Project Setup</Heading>
```

### Level 2

```tsx
<Heading level={2}>Configuration</Heading>
```

### Level 3 and Level 4

```tsx
<Heading level={3}>Options</Heading>
<Heading level={4}>Advanced Settings</Heading>
```

### Custom Color Without Uppercase

```tsx
<Heading level={1} color="#e74c3c" uppercase={false}>
  Custom Heading
</Heading>
```
