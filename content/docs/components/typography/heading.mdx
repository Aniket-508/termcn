## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/heading.json
```

## Usage

```tsx
import { Heading } from "@/components/ui/heading";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Heading {...demoProps} />;
}
```

## Props

| Prop        | Type           | Default     | Description                                              |
| ----------- | -------------- | ----------- | -------------------------------------------------------- |
| `level`     | `HeadingLevel` | `1`         | -                                                        |
| `children`  | `ReactNode`    | `required`  | -                                                        |
| `color`     | `string`       | `undefined` | -                                                        |
| `prefix1`   | `string`       | `"██ "`     | Prefix symbol for level 1 headings. Default: '██ '       |
| `prefix2`   | `string`       | `"▌ "`      | Prefix symbol for level 2 headings. Default: '▌ '        |
| `prefix3`   | `string`       | `"› "`      | Prefix symbol for level 3 headings. Default: '› '        |
| `uppercase` | `boolean`      | `true`      | Whether to uppercase level 1 heading text. Default: true |
