## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/gradient.json
```

## Usage

```tsx
import { Gradient } from "@/components/ui/gradient";

export function Example() {
  return <Gradient colors={["#ff6b6b", "#4ecdc4"]}>Hello Gradient</Gradient>;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type       | Default    | Description                               |
| ---------- | ---------- | ---------- | ----------------------------------------- |
| `children` | `string`   | `required` | Text string to apply the gradient to      |
| `colors`   | `string[]` | `required` | Array of hex color stops for the gradient |
| `bold`     | `boolean`  | `false`    | Whether to render the text in bold        |

## Examples

### Basic Usage

```tsx
<Gradient colors={["#ee5a24", "#f9ca24"]}>Sunset Text</Gradient>
```

### Three-Color Gradient

```tsx
<Gradient colors={["#7f8c8d", "#2ecc71", "#3498db"]}>
  Multi-stop gradient
</Gradient>
```

### Bold Gradient

```tsx
<Gradient colors={["#6c5ce7", "#fd79a8"]} bold>
  Bold Gradient
</Gradient>
```
