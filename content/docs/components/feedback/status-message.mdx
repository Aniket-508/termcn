## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/status-message.json
```

## Usage

```tsx
import { StatusMessage } from "@/components/ui/status-message";

export function Example() {
  return <StatusMessage variant="success">All tests passed.</StatusMessage>;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type            | Default     | Description                                                                 |
| ---------- | --------------- | ----------- | --------------------------------------------------------------------------- |
| `variant`  | `StatusVariant` | `"info"`    | Status type: "success", "error", "warning", "info", "loading", or "pending" |
| `children` | `ReactNode`     | `required`  | Message content                                                             |
| `icon`     | `string`        | `undefined` | Custom icon character, overrides variant default                            |

## Examples

### Error Status

```tsx
<StatusMessage variant="error">Connection to database failed.</StatusMessage>
```

### Loading Status

```tsx
<StatusMessage variant="loading">Fetching remote config...</StatusMessage>
```

### Warning with Custom Icon

```tsx
<StatusMessage variant="warning" icon="⏳">
  Rate limit approaching.
</StatusMessage>
```
