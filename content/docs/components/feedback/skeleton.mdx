## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/skeleton.json
```

## Usage

```tsx
import { Skeleton } from "@/components/ui/skeleton";

export function Example() {
  return <Skeleton width={30} height={3} />;
}
```

## Dependencies

### Registry Dependencies

- `use-animation`

### npm Dependencies

- `ink`

## Props

| Prop       | Type      | Default | Description                        |
| ---------- | --------- | ------- | ---------------------------------- |
| `width`    | `number`  | `20`    | Character width of the placeholder |
| `height`   | `number`  | `1`     | Number of rows                     |
| `animated` | `boolean` | `true`  | Enable shimmer animation           |

## Examples

### Single Line

```tsx
<Skeleton width={24} />
```

### Multi-Line Block

```tsx
<Skeleton width={40} height={4} />
```

### Static (No Animation)

```tsx
<Skeleton width={20} height={2} animated={false} />
```
