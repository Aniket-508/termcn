## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/toast.json
```

## Usage

```tsx
import { Toast } from "@/components/ui/toast";

export function Example() {
  return (
    <Toast
      variant="success"
      message="Settings saved successfully."
      duration={5000}
      onDismiss={() => console.log("toast closed")}
    />
  );
}
```

## Props

| Prop        | Type           | Default     | Description                                      |
| ----------- | -------------- | ----------- | ------------------------------------------------ |
| `message`   | `string`       | `required`  | Text content of the toast                        |
| `variant`   | `ToastVariant` | `"info"`    | Visual style variant                             |
| `duration`  | `number`       | `3000`      | Auto-dismiss delay in milliseconds               |
| `onDismiss` | `() => void`   | `undefined` | Callback fired when the toast is dismissed       |
| `icon`      | `string`       | `undefined` | Custom icon character, overrides variant default |

## Examples

### Info Toast

