## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/toast.json
```

## Usage

```tsx
import { Toast } from "@/components/ui/toast";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Toast {...demoProps} />;
}
```
