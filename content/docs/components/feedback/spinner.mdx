## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/spinner.json
```

## Usage

```tsx
import { Spinner } from "@/components/ui/spinner";

export function Example() {
  return <Spinner style="dots" label="Loading components" />;
}
```

## Dependencies

### Registry Dependencies

- `theme-provider`

### npm Dependencies

- `ink`

## Props

| Prop     | Type           | Default     | Description                              |
| -------- | -------------- | ----------- | ---------------------------------------- |
| `style`  | `SpinnerStyle` | `"dots"`    | Animation style preset                   |
| `label`  | `string`       | `undefined` | Text displayed next to the spinner       |
| `color`  | `string`       | `undefined` | Override the spinner color               |
| `fps`    | `number`       | `12`        | Animation speed in frames per second     |
| `frames` | `string[]`     | `undefined` | Custom animation frames, overrides style |

## Examples

### With Label

```tsx
<Spinner label="Installing dependencies" />
```

### Different Styles

```tsx
<Spinner style="earth" label="Deploying" />
```

### Custom Speed and Color

```tsx
<Spinner style="arc" fps={6} color="#f59e0b" label="Compiling" />
```

### Custom Frames

```tsx
<Spinner frames={["🌑", "🌒", "🌓", "🌔", "🌕"]} label="Processing" />
```
