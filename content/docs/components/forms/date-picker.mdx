## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/date-picker.json
```

## Usage

```tsx
import { DatePicker } from "@/components/ui/date-picker";

export function Example() {
  return (
    <DatePicker
      label="Birth date"
      value={new Date(2000, 0, 1)}
      onChange={(date) => console.log("Selected:", date)}
      onSubmit={(date) => console.log("Submitted:", date)}
    />
  );
}
```

## Props

| Prop        | Type                   | Default     | Description                                          |
| ----------- | ---------------------- | ----------- | ---------------------------------------------------- |
| `value`     | `Date`                 | `undefined` | Controlled date value                                |
| `onChange`  | `(date: Date) => void` | `undefined` | Callback fired when the selected date changes        |
| `onSubmit`  | `(date: Date) => void` | `undefined` | Callback fired when the user presses Enter to submit |
| `label`     | `string`               | `undefined` | Label displayed above the picker                     |
| `minDate`   | `Date`                 | `undefined` | Earliest selectable date                             |
| `maxDate`   | `Date`                 | `undefined` | Latest selectable date                               |
| `autoFocus` | `boolean`              | `false`     | Whether the picker receives focus on mount           |
| `id`        | `string`               | `undefined` | Focus ID for use with `useFocus`                     |

## Examples

### With Label

```tsx
<DatePicker label="Start date" onSubmit={(date) => console.log(date)} />
```

### Constrained Range

```tsx
<DatePicker
  label="Appointment"
  minDate={new Date(2025, 0, 1)}
  maxDate={new Date(2025, 11, 31)}
  onSubmit={(date) => scheduleAppointment(date)}
/>
```

### Auto-Focused

```tsx
<DatePicker
  autoFocus
  onChange={(date) => setSelectedDate(date)}
  onSubmit={(date) => confirm(date)}
/>
```
