## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/date-picker.json
```

## Usage

```tsx
import { DatePicker } from "@/components/ui/date-picker";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <DatePicker {...demoProps} />;
}
```
