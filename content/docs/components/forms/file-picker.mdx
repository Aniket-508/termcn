## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/file-picker.json
```

## Usage

```tsx
import { FilePicker } from "@/components/ui/file-picker";

export function Example() {
  return (
    <FilePicker
      label="Select a config file"
      extensions={[".json", ".yaml", ".yml"]}
      onSubmit={(path) => console.log("Selected:", path)}
    />
  );
}
```

## Props

| Prop         | Type                     | Default         | Description                                       |
| ------------ | ------------------------ | --------------- | ------------------------------------------------- |
| `value`      | `string`                 | `undefined`     | Controlled selected file path                     |
| `onChange`   | `(path: string) => void` | `undefined`     | Callback fired when a file is selected            |
| `onSubmit`   | `(path: string) => void` | `undefined`     | Callback fired when the user confirms a selection |
| `label`      | `string`                 | `undefined`     | Label displayed above the picker                  |
| `startDir`   | `string`                 | `process.cwd()` | Starting directory for the file browser           |
| `extensions` | `string[]`               | `undefined`     | Show only files with these extensions             |
| `dirsOnly`   | `boolean`                | `false`         | Show only directories                             |
| `autoFocus`  | `boolean`                | `false`         | Whether the picker receives focus on mount        |
| `id`         | `string`                 | `undefined`     | Focus ID for use with `useFocus`                  |
| `width`      | `number`                 | `50`            | Width of the file browser panel in columns        |
| `maxVisible` | `number`                 | `8`             | Maximum number of entries visible at once         |

## Examples

### Directory-Only Mode

```tsx
<FilePicker
  label="Choose output directory"
  dirsOnly
  onSubmit={(dir) => setOutputDir(dir)}
/>
```

### Filter by Extension

```tsx
<FilePicker
  label="Pick a TypeScript file"
  extensions={[".ts", ".tsx"]}
  startDir="./src"
  onChange={(path) => console.log(path)}
/>
```

### Custom Width and Visible Count

```tsx
<FilePicker
  label="Browse project"
  width={70}
  maxVisible={12}
  autoFocus
  onSubmit={(path) => openFile(path)}
/>
```
