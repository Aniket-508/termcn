## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/time-picker.json
```

## Usage

```tsx
import { TimePicker } from "@/components/ui/time-picker";

export function Example() {
  return (
    <TimePicker
      label="Meeting time"
      value={{ hours: 14, minutes: 30 }}
      onChange={(time) => console.log("Changed:", time)}
      onSubmit={(time) => console.log("Submitted:", time)}
    />
  );
}
```

## Props

| Prop        | Type                                                 | Default     | Description                                          |
| ----------- | ---------------------------------------------------- | ----------- | ---------------------------------------------------- |
| `value`     | `{ hours: number; minutes: number }`                 | `undefined` | Controlled time value                                |
| `onChange`  | `(time: { hours: number; minutes: number }) => void` | `undefined` | Callback fired when the selected time changes        |
| `onSubmit`  | `(time: { hours: number; minutes: number }) => void` | `undefined` | Callback fired when the user presses Enter to submit |
| `label`     | `string`                                             | `undefined` | Label displayed above the picker                     |
| `format`    | `12 \| 24`                                           | `24`        | 12-hour or 24-hour clock format                      |
| `autoFocus` | `boolean`                                            | `false`     | Whether the picker receives focus on mount           |
| `id`        | `string`                                             | `undefined` | Focus ID for use with `useFocus`                     |

## Examples

### 12-Hour Format

```tsx
<TimePicker
  label="Alarm"
  format={12}
  value={{ hours: 7, minutes: 0 }}
  onSubmit={(time) => setAlarm(time)}
/>
```

### Auto-Focused

```tsx
<TimePicker
  autoFocus
  label="Start time"
  onChange={(time) => setStartTime(time)}
/>
```

### No Label

```tsx
<TimePicker
  format={24}
  value={{ hours: 18, minutes: 45 }}
  onSubmit={(time) => scheduleEvent(time)}
/>
```
