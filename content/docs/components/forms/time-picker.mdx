## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/time-picker.json
```

## Usage

```tsx
import { TimePicker } from "@/components/ui/time-picker";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <TimePicker {...demoProps} />;
}
```

## Props

| Prop        | Type                                                              | Default     | Description                       |
| ----------- | ----------------------------------------------------------------- | ----------- | --------------------------------- |
| `value`     | `&#123; hours: number; minutes: number &#125;`                    | `undefined` | -                                 |
| `onChange`  | `(time: &#123; hours: number; minutes: number &#125;) =&gt; void` | `undefined` | -                                 |
| `onSubmit`  | `(time: &#123; hours: number; minutes: number &#125;) =&gt; void` | `undefined` | -                                 |
