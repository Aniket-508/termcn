## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/stopwatch.json
```

## Usage

```tsx
import { Stopwatch } from "@/components/ui/stopwatch";

export function Example() {
  return <Stopwatch autoStart showLaps color="green" />;
}
```

## Dependencies

### Registry Dependencies

- `use-input`
- `theme-provider`

### npm Dependencies

- `ink`

## Props

| Prop        | Type      | Default     | Description                               |
| ----------- | --------- | ----------- | ----------------------------------------- |
| `autoStart` | `boolean` | `false`     | Start the stopwatch immediately on mount  |
| `color`     | `string`  | `undefined` | Custom color for the elapsed time display |
| `showLaps`  | `boolean` | `true`      | Whether to display the lap times list     |

## Examples

### Auto-starting stopwatch

```tsx
<Stopwatch autoStart />
```

### Without lap tracking

```tsx
<Stopwatch showLaps={false} />
```

### Custom color

```tsx
<Stopwatch color="cyan" autoStart showLaps />
```
