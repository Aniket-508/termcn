## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/error-boundary.json
```

## Usage

```tsx
import { ErrorBoundary } from "@/components/ui/error-boundary";
import { Text } from "ink";

export function Example() {
  return (
    <ErrorBoundary
      title="Application Error"
      onError={(error, info) => console.error(error, info.componentStack)}
    >
      <Text>Your app content here</Text>
    </ErrorBoundary>
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop       | Type                                                       | Default     | Description                                         |
| ---------- | ---------------------------------------------------------- | ----------- | --------------------------------------------------- |
| `children` | `ReactNode`                                                | `required`  | The component tree to wrap with error protection    |
| `fallback` | `ReactNode`                                                | `undefined` | Custom fallback UI rendered when an error is caught |
| `onError`  | `(error: Error, info: { componentStack: string }) => void` | `undefined` | Callback fired when an error is caught              |
| `title`    | `string`                                                   | `"Error"`   | Heading displayed in the default error panel        |

## Examples

### Basic wrapper

```tsx
<ErrorBoundary>
  <MyComponent />
</ErrorBoundary>
```

### Custom fallback UI

```tsx
<ErrorBoundary fallback={<Text color="red">Something went wrong.</Text>}>
  <RiskyComponent />
</ErrorBoundary>
```

### With error logging and title

```tsx
<ErrorBoundary
  title="Plugin Crash"
  onError={(error) => logger.error("Plugin failed", error)}
>
  <PluginHost />
</ErrorBoundary>
```
