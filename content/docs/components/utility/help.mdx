## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/help.json
```

## Usage

```tsx
import { Help } from "@/components/ui/help";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Help {...demoProps} />;
}
```
