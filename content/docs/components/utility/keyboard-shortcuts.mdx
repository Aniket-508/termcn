## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/keyboard-shortcuts.json
```

## Usage

```tsx
import { KeyboardShortcuts } from "@/components/ui/keyboard-shortcuts";

export function Example() {
  return (
    <KeyboardShortcuts
      title="Editor Shortcuts"
      shortcuts={[
        { key: "Ctrl+S", description: "Save file" },
        { key: "Ctrl+Z", description: "Undo" },
        { key: "Ctrl+F", description: "Find" },
      ]}
    />
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop        | Type         | Default     | Description                                                                  |
| ----------- | ------------ | ----------- | ---------------------------------------------------------------------------- |
| `shortcuts` | `Shortcut[]` | `required`  | Array of shortcut objects with `key`, `description`, and optional `category` |
| `columns`   | `number`     | `1`         | Number of columns to display shortcuts in a grid layout                      |
| `title`     | `string`     | `undefined` | Title displayed at the top of the shortcuts panel                            |

## Examples

### Multi-column layout

```tsx
<KeyboardShortcuts
  columns={2}
  shortcuts={[
    { key: "j", description: "Move down" },
    { key: "k", description: "Move up" },
    { key: "h", description: "Move left" },
    { key: "l", description: "Move right" },
  ]}
/>
```

### Grouped by category

```tsx
<KeyboardShortcuts
  title="App Shortcuts"
  shortcuts={[
    { key: "Ctrl+N", description: "New file", category: "File" },
    { key: "Ctrl+O", description: "Open file", category: "File" },
    { key: "Ctrl+C", description: "Copy", category: "Edit" },
    { key: "Ctrl+V", description: "Paste", category: "Edit" },
  ]}
/>
```

### Simple list without title

```tsx
<KeyboardShortcuts
  shortcuts={[
    { key: "q", description: "Quit" },
    { key: "?", description: "Show help" },
  ]}
/>
```
