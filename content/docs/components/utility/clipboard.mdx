## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/clipboard.json
```

## Usage

```tsx
import { Clipboard } from "@/components/ui/clipboard";

export function Example() {
  return (
    <Clipboard
      value="npm install termcn"
      label="Install command"
      successMessage="Copied!"
      timeout={2000}
    />
  );
}
```

## Dependencies

### Registry Dependencies

- `theme-provider`
- `use-clipboard`

### npm Dependencies

- `ink`

## Props

| Prop             | Type     | Default     | Description                                             |
| ---------------- | -------- | ----------- | ------------------------------------------------------- |
| `value`          | `string` | `required`  | The text value to copy to the clipboard                 |
| `label`          | `string` | `undefined` | Optional label displayed above the clipboard field      |
| `successMessage` | `string` | `"Copied!"` | Message shown after a successful copy                   |
| `timeout`        | `number` | `2000`      | Duration in ms to show the success message before reset |

## Examples

### Basic copy field

```tsx
<Clipboard value="sk-abc123xyz" />
```

### With label and custom message

```tsx
<Clipboard
  value="https://example.com/invite/abc"
  label="Share link"
  successMessage="Link copied!"
/>
```

### Short feedback timeout

```tsx
<Clipboard
  value="export API_KEY=abc123"
  label="Environment variable"
  timeout={1000}
/>
```
