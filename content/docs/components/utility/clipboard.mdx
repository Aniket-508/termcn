## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/clipboard.json
```

## Usage

```tsx
import { Clipboard } from "@/components/ui/clipboard";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Clipboard {...demoProps} />;
}
```
