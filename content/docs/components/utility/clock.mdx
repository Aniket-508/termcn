## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/clock.json
```

## Usage

```tsx
import { Clock } from "@/components/ui/clock";

export function Example() {
  return (
    <Clock format="12h" showSeconds showDate timezone="America/New_York" />
  );
}
```

## Props

| Prop          | Type             | Default     | Description                                      |
| ------------- | ---------------- | ----------- | ------------------------------------------------ |
| `format`      | `"12h" \| "24h"` | `"24h"`     | Time format to use                               |
| `showSeconds` | `boolean`        | `true`      | Whether to display seconds                       |
| `showDate`    | `boolean`        | `false`     | Show the current date above the time             |
| `timezone`    | `string`         | `undefined` | IANA timezone string (e.g. `"America/New_York"`) |
| `color`       | `string`         | `undefined` | Custom color for the time display                |
| `size`        | `"sm" \| "lg"`   | `"sm"`      | Display size; `"lg"` uses box-drawing digit art  |

## Examples

### 24-hour clock without seconds

```tsx
<Clock format="24h" showSeconds={false} />
```

### Large display with date

```tsx
<Clock size="lg" showDate format="12h" />
```

### Timezone-specific clock

```tsx
<Clock timezone="Asia/Tokyo" format="24h" color="cyan" />
```
