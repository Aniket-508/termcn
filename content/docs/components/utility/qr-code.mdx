## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/qr-code.json
```

## Usage

```tsx
import { QRCode } from "@/components/ui/qr-code";

export function Example() {
  return <QRCode value="https://termcn.dev" size="md" label="Scan to visit" />;
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop    | Type                   | Default     | Description                                                      |
| ------- | ---------------------- | ----------- | ---------------------------------------------------------------- |
| `value` | `string`               | `required`  | The text or URL to encode (max ~17 bytes for V1 QR)              |
| `size`  | `"sm" \| "md" \| "lg"` | `"md"`      | Render size: `"sm"` uses half-blocks, `"lg"` doubles each module |
| `color` | `string`               | `undefined` | Custom color for the QR modules                                  |
| `label` | `string`               | `undefined` | Label displayed below the QR code                                |

## Examples

### Small compact QR

```tsx
<QRCode value="https://example.com" size="sm" />
```

### Large QR with custom color

```tsx
<QRCode value="HELLO" size="lg" color="cyan" label="Scan me" />
```

### Numeric data

```tsx
<QRCode value="1234567890" size="md" label="Order #1234567890" />
```
