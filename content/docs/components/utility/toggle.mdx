```bash

```

## Usage

```tsx
import { Toggle } from "@/components/ui/toggle";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Toggle {...demoProps} />;
}
```

## Props

| Prop            | Type                                                                                            | Default     | Description                             |
| --------------- | ----------------------------------------------------------------------------------------------- | ----------- | --------------------------------------- |
| `checked`       | `boolean`                                                                                       | `undefined` | -                                       |
| `onChange`      | `(checked: boolean) =&gt; void`                                                                 | `undefined` | -                                       |
| `label`         | `string`                                                                                        | `undefined` | -                                       |
| `onLabel`       | `string`                                                                                        | `"ON"`      | -                                       |
| `offLabel`      | `string`                                                                                        | `"OFF"`     | -                                       |
| `id`            | `string`                                                                                        | `undefined` | -                                       |
| `disabled`      | `boolean`                                                                                       | `false`     | -                                       |
| `checkedIcon`   | `string`                                                                                        | `"●"`       | Icon shown when checked. Default: '●'   |
| `uncheckedIcon` | `string`                                                                                        | `"○"`       | Icon shown when unchecked. Default: '○' |
| `borderStyle`   | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"round"`   | Border style. Default: 'round'          |
| `paddingX`      | `number`                                                                                        | `1`         | Horizontal padding. Default: 1          |
