## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/menu.json
```

## Usage

```tsx
import { Menu } from "@/components/ui/menu";

export function Example() {
  return (
    <Menu
      title="File"
      items={[
        { key: "new", label: "New File", icon: "📄", shortcut: "⌘N" },
        { key: "open", label: "Open...", icon: "📂", shortcut: "⌘O" },
        { key: "sep1", label: "", separator: true },
        { key: "save", label: "Save", shortcut: "⌘S" },
        { key: "export", label: "Export", disabled: true },
      ]}
      onSelect={(item) => console.log(item.key)}
    />
  );
}
```

## Props

| Prop       | Type                       | Default     | Description                                         |
| ---------- | -------------------------- | ----------- | --------------------------------------------------- |
| `items`    | `MenuItem[]`               | `required`  | Array of menu items to render                       |
| `onSelect` | `(item: MenuItem) => void` | `undefined` | Callback fired when a non-disabled item is selected |
| `title`    | `string`                   | `undefined` | Optional title displayed above the menu             |

### MenuItem

| Prop        | Type         | Default     | Description                                       |
| ----------- | ------------ | ----------- | ------------------------------------------------- |
| `key`       | `string`     | `required`  | Unique identifier for the item                    |
| `label`     | `string`     | `required`  | Display text for the menu item                    |
| `icon`      | `string`     | `undefined` | Icon character shown before the label             |
| `shortcut`  | `string`     | `undefined` | Keyboard shortcut hint shown on the right         |
| `disabled`  | `boolean`    | `undefined` | When true, the item cannot be selected or focused |
| `separator` | `boolean`    | `undefined` | When true, renders a horizontal divider line      |
| `children`  | `MenuItem[]` | `undefined` | Nested submenu items opened with the right arrow  |

## Examples

### With nested submenus

```tsx
<Menu
  items={[
    {
      key: "view",
      label: "View",
      children: [
        { key: "zoom-in", label: "Zoom In", shortcut: "⌘+" },
        { key: "zoom-out", label: "Zoom Out", shortcut: "⌘-" },
      ],
    },
    { key: "settings", label: "Settings", icon: "⚙" },
  ]}
  onSelect={(item) => console.log(item.key)}
/>
```

### Minimal menu without title

```tsx
<Menu
```
