## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/breadcrumb.json
```

## Usage

```tsx
import { Breadcrumb } from "@/components/ui/breadcrumb";

export function Example() {
  return (
    <Breadcrumb
      items={[
        { key: "home", label: "Home", onSelect: () => console.log("home") },
        { key: "docs", label: "Docs", onSelect: () => console.log("docs") },
        { key: "components", label: "Components" },
      ]}
      activeKey="components"
      separator="›"
    />
  );
}
```

## Props

| Prop        | Type               | Default     | Description                                                |
| ----------- | ------------------ | ----------- | ---------------------------------------------------------- |
| `items`     | `BreadcrumbItem[]` | `required`  | Array of breadcrumb items to display in the trail          |
| `separator` | `string`           | `"›"`       | Character displayed between breadcrumb items               |
| `activeKey` | `string`           | `undefined` | Key of the active item; defaults to the last item if unset |

### BreadcrumbItem

| Prop       | Type         | Default     | Description                              |
| ---------- | ------------ | ----------- | ---------------------------------------- |
| `label`    | `string`     | `required`  | Display text for the breadcrumb item     |
| `key`      | `string`     | `required`  | Unique identifier for the item           |
| `onSelect` | `() => void` | `undefined` | Callback fired when navigating backwards |

## Examples

### Custom separator

```tsx
<Breadcrumb
  items={[
    { key: "root", label: "Root" },
    { key: "src", label: "src" },
    { key: "index", label: "index.ts" },
  ]}
  separator="/"
/>
```

### Two-level path

```tsx
<Breadcrumb
  items={[
    { key: "settings", label: "Settings", onSelect: () => goBack() },
    { key: "profile", label: "Profile" },
  ]}
  activeKey="profile"
/>
```
