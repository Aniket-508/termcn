## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/json.json
```

## Usage

```tsx
import { JSONView } from "@/components/ui/json";

export function Example() {
  return (
    <JSONView
      label="package.json"
      data={{
        name: "my-cli",
        version: "1.0.0",
        dependencies: {
          ink: "^5.0.0",
          react: "^18.0.0",
        },
      }}
    />
  );
}
```

## Dependencies

### Registry Dependencies

- `theme-provider`

### npm Dependencies

- `ink`

## Props

| Prop        | Type      | Default     | Description                                  |
| ----------- | --------- | ----------- | -------------------------------------------- |
| `data`      | `unknown` | `required`  | The JSON-serializable data to display        |
| `indent`    | `number`  | `2`         | Number of spaces per indentation level       |
| `collapsed` | `boolean` | `false`     | Whether all nodes start in a collapsed state |
| `label`     | `string`  | `undefined` | Label displayed above the JSON tree          |

## Examples

### Collapsed by default

```tsx
<JSONView
  collapsed
  data={{
    users: [
      { name: "Alice", age: 30 },
      { name: "Bob", age: 25 },
    ],
    meta: { total: 2 },
  }}
/>
```

### Custom indentation

```tsx
<JSONView indent={4} data={["react", "ink", "chalk", "yoga-layout"]} />
```

### Simple key-value data

```tsx
<JSONView
  label="Environment"
  data={{
    NODE_ENV: "production",
    PORT: 3000,
    DEBUG: false,
  }}
/>
```
