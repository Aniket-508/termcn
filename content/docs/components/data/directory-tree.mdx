```bash
npx shadcn@latest add https://termcn.vercel.app/r/directory-tree.json
```

## Usage

```tsx
import { DirectoryTree } from "@/components/ui/directory-tree";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <DirectoryTree {...demoProps} />;
}
```

## Props

| Prop         | Type                        | Default         | Description                                         |
| ------------ | --------------------------- | --------------- | --------------------------------------------------- |
| `rootPath`   | `string`                    | `process.cwd()` | Root path to display. Default: process.cwd()        |
| `onSelect`   | `(path: string) =&gt; void` | `undefined`     | -                                                   |
| `maxDepth`   | `number`                    | `2`             | Max depth to show initially. Default: 2             |
| `showHidden` | `boolean`                   | `false`         | Show hidden files (starting with .). Default: false |
| `label`      | `string`                    | `undefined`     | -                                                   |
