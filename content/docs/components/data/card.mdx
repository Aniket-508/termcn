## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/card.json
```

## Usage

```tsx
import { Card } from "@/components/ui/card";
import { Text } from "ink";

export function Example() {
  return (
    <Card title="System Info" subtitle="Local machine">
      <Text>CPU: 4 cores @ 3.2 GHz</Text>
      <Text>Memory: 16 GB</Text>
    </Card>
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop                | Type                                                                                            | Default     | Description                                |
| ------------------- | ----------------------------------------------------------------------------------------------- | ----------- | ------------------------------------------ |
| `title`             | `string`                                                                                        | `undefined` | Title displayed at the top of the card     |
| `subtitle`          | `string`                                                                                        | `undefined` | Subtitle displayed below the title         |
| `children`          | `ReactNode`                                                                                     | `required`  | Body content rendered inside the card      |
| `footer`            | `ReactNode`                                                                                     | `undefined` | Content rendered below the footer divider  |
| `borderColor`       | `string`                                                                                        | `undefined` | Override border color (uses theme default) |
| `width`             | `number`                                                                                        | `undefined` | Fixed width of the card in columns         |
| `borderStyle`       | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"round"`   | Border style for the card outline          |
| `paddingX`          | `number`                                                                                        | `1`         | Horizontal padding inside the card         |
| `paddingY`          | `number`                                                                                        | `0`         | Vertical padding inside the card           |
| `footerDividerChar` | `string`                                                                                        | `"─"`       | Character used for the footer divider      |

## Examples

### Card with footer

```tsx
<Card title="Deploy Status" footer={<Text color="green">✔ Deployed</Text>}>
  <Text>Region: us-east-1</Text>
  <Text>Instances: 3</Text>
</Card>
```

### Custom border style

```tsx
<Card title="Alerts" borderStyle="double" borderColor="red">
  <Text>3 critical issues found</Text>
</Card>
```

### Minimal card without header

```tsx
<Card paddingX={2} paddingY={1} width={40}>
  <Text>A simple content-only card.</Text>
</Card>
```
