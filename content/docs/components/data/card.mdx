## Installation

```bash

```

## Usage

```tsx
import { Card } from "@/components/ui/card";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <Card {...demoProps} />;
}
```

## Props

| Prop                | Type                                                                                            | Default     | Description                                         |
| ------------------- | ----------------------------------------------------------------------------------------------- | ----------- | --------------------------------------------------- |
| `title`             | `string`                                                                                        | `undefined` | -                                                   |
| `subtitle`          | `string`                                                                                        | `undefined` | -                                                   |
| `children`          | `ReactNode`                                                                                     | `required`  | -                                                   |
| `footer`            | `ReactNode`                                                                                     | `undefined` | -                                                   |
| `borderColor`       | `string`                                                                                        | `undefined` | -                                                   |
| `width`             | `number`                                                                                        | `undefined` | -                                                   |
| `borderStyle`       | `\| "single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"round"`   | Border style. Default: 'round'                      |
| `paddingX`          | `number`                                                                                        | `1`         | Horizontal padding. Default: 1                      |
| `paddingY`          | `number`                                                                                        | `0`         | Vertical padding. Default: 0                        |
| `footerDividerChar` | `string`                                                                                        | `"─"`       | Character used for the footer divider. Default: '─' |
