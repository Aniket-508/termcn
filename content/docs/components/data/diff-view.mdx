## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/diff-view.json
```

## Usage

```tsx
import { DiffView } from "@/components/ui/diff-view";

export function Example() {
  return (
    <DiffView
      filename="config.ts"
      oldText={"const port = 3000;\nconst host = 'localhost';"}
      newText={
        "const port = 8080;\nconst host = 'localhost';\nconst debug = true;"
      }
    />
  );
}
```

## Props

| Prop              | Type                               | Default     | Description                                |
| ----------------- | ---------------------------------- | ----------- | ------------------------------------------ |
| `oldText`         | `string`                           | `required`  | Original text content to diff against      |
| `newText`         | `string`                           | `required`  | Updated text content                       |
| `filename`        | `string`                           | `undefined` | Optional filename displayed as a header    |
| `language`        | `string`                           | `undefined` | Language hint for the content              |
| `mode`            | `"unified" \| "split" \| "inline"` | `"unified"` | Diff display mode                          |
| `context`         | `number`                           | `3`         | Number of context lines around each change |
| `showLineNumbers` | `boolean`                          | `false`     | Show line numbers in the gutter            |

## Examples

### Split mode with line numbers

```tsx
<DiffView
  mode="split"
  showLineNumbers
  oldText={"function add(a, b) {\n  return a + b;\n}"}
  newText={"function add(a: number, b: number): number {\n  return a + b;\n}"}
/>
```

### Inline mode

```tsx
<DiffView
  mode="inline"
  filename="README.md"
  oldText={"# My Project\nA simple tool."}
  newText={"# My Project\nA powerful CLI tool.\n\n## Features\n- Fast"}
/>
```

### With extra context lines

```tsx
<DiffView
  context={5}
  showLineNumbers
  oldText={"line 1\nline 2\nline 3\nold line\nline 5"}
  newText={"line 1\nline 2\nline 3\nnew line\nline 5"}
/>
```
