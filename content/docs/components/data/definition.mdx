## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/definition.json
```

## Usage

```tsx
import { Definition } from "@/components/ui/definition";

export function Example() {
  return (
    <Definition
      items={[
        { term: "CLI", description: "Command Line Interface" },
        { term: "TUI", description: "Terminal User Interface" },
        { term: "REPL", description: "Read-Eval-Print Loop" },
      ]}
    />
  );
}
```

## Dependencies

### npm Dependencies

- `ink`

## Props

| Prop        | Type               | Default     | Description                                |
| ----------- | ------------------ | ----------- | ------------------------------------------ |
| `items`     | `DefinitionItem[]` | `required`  | Array of term/description pairs to display |
| `termColor` | `string`           | `undefined` | Override color for the term labels         |

### DefinitionItem

| Prop          | Type     | Description             |
| ------------- | -------- | ----------------------- |
| `term`        | `string` | The term or label       |
| `description` | `string` | The definition or value |

## Examples

### Custom term color

```tsx
<Definition
  termColor="cyan"
  items={[
    { term: "Node.js", description: "JavaScript runtime built on V8" },
    {
      term: "Deno",
      description: "Secure runtime for JavaScript and TypeScript",
    },
  ]}
/>
```

### Single item

```tsx
<Definition items={[{ term: "Version", description: "1.4.2" }]} />
```
