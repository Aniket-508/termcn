## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/checkbox.json
```

## Usage

```tsx
import { Checkbox } from "@/components/ui/checkbox";

export function Example() {
  return (
    <Checkbox
      checked
      label="Enable telemetry"
      onChange={(checked) => console.log("Checked:", checked)}
    />
  );
}
```

## Props

| Prop                | Type                         | Default     | Description                                   |
| ------------------- | ---------------------------- | ----------- | --------------------------------------------- |
| `checked`           | `boolean`                    | `undefined` | Whether the checkbox is checked               |
| `onChange`          | `(checked: boolean) => void` | `undefined` | Callback fired when the checked state changes |
| `label`             | `string`                     | `undefined` | Text label displayed next to the checkbox     |
| `indeterminate`     | `boolean`                    | `false`     | Show an indeterminate (partial) state icon    |
| `disabled`          | `boolean`                    | `false`     | Prevent interaction and dim the checkbox      |
| `id`                | `string`                     | `undefined` | Focus identifier for `useFocus`               |
| `checkedIcon`       | `string`                     | `"■"`       | Character shown when checked                  |
| `uncheckedIcon`     | `string`                     | `"□"`       | Character shown when unchecked                |
| `indeterminateIcon` | `string`                     | `"▪"`       | Character shown when indeterminate            |

## Examples

### Disabled Checkbox

```tsx
<Checkbox label="Locked option" disabled />
```

### Indeterminate State

```tsx
<Checkbox label="Select all" indeterminate />
```

### Custom Icons

```tsx
<Checkbox label="Accept terms" checkedIcon="✓" uncheckedIcon="✗" />
```
