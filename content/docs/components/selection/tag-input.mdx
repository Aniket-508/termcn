## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tag-input.json
```

## Usage

```tsx
import { TagInput } from "@/components/ui/tag-input";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <TagInput {...demoProps} />;
}
```
