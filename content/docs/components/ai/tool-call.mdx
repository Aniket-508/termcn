## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tool-call.json
```

## Usage

```tsx
import { ToolCall } from "@/components/ui/tool-call";

export function Example() {
  return (
    <ToolCall
      name="search_codebase"
      args={{ query: "handleSubmit", maxResults: 10 }}
      status="success"
      result="Found 3 matches in src/components/"
      duration={245}
    />
  );
}
```

## Props

| Prop               | Type                      | Default     | Description                                                         |
| ------------------ | ------------------------- | ----------- | ------------------------------------------------------------------- |
| `name`             | `string`                  | `required`  | Name of the tool being invoked.                                     |
| `args`             | `Record<string, unknown>` | `undefined` | Key-value arguments passed to the tool.                             |
| `status`           | `ToolCallStatus`          | `required`  | Current state: `"pending"`, `"running"`, `"success"`, or `"error"`. |
| `result`           | `unknown`                 | `undefined` | Return value from the tool, shown when expanded.                    |
| `duration`         | `number`                  | `undefined` | Execution time in milliseconds, displayed next to the name.         |
| `collapsible`      | `boolean`                 | `true`      | Whether the args/result section can be toggled with Enter or Space. |
| `defaultCollapsed` | `boolean`                 | `true`      | Whether the detail section starts collapsed.                        |

## Examples

### Running state

```tsx
<ToolCall
  name="fetch_url"
  args={{ url: "https://api.example.com/data" }}
  status="running"
/>
```

### Completed with result

```tsx
<ToolCall
  name="read_file"
  args={{ path: "package.json" }}
  status="success"
  result={{ name: "my-app", version: "1.0.0" }}
  duration={12}
  defaultCollapsed={false}
/>
```

### Error state

```tsx
<ToolCall
  name="execute_command"
  args={{ command: "npm test" }}
  status="error"
  result="Process exited with code 1"
  duration={4320}
/>
```

### Non-collapsible pending

```tsx
<ToolCall name="lint_files" status="pending" collapsible={false} />
```
