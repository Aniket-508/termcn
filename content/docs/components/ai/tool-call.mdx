```bash
npx shadcn@latest add https://termcn.vercel.app/r/tool-call.json
```

## Usage

```tsx
import { ToolCall } from "@/components/ui/tool-call";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <ToolCall {...demoProps} />;
```

## Props

| Prop               | Type                            | Default     | Description |
| ------------------ | ------------------------------- | ----------- | ----------- |
| `name`             | `string`                        | `required`  | -           |
| `args`             | `Record&lt;string, unknown&gt;` | `undefined` | -           |
| `status`           | `ToolCallStatus`                | `required`  | -           |
| `result`           | `unknown`                       | `undefined` | -           |
| `duration`         | `number`                        | `undefined` | -           |
| `collapsible`      | `boolean`                       | `true`      | -           |
| `defaultCollapsed` | `boolean`                       | `true`      | -           |
