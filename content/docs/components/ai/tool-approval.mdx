## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tool-approval.json
```

## Usage

```tsx
import { ToolApproval } from "@/components/ui/tool-approval";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <ToolApproval {...demoProps} />;
}
```

## Props

| Prop            | Type                            | Default     | Description                  |
| --------------- | ------------------------------- | ----------- | ---------------------------- |
| `name`          | `string`                        | `required`  | -                            |
| `description`   | `string`                        | `undefined` | -                            |
| `args`          | `Record&lt;string, unknown&gt;` | `undefined` | -                            |
| `risk`          | `RiskLevel`                     | `"low"`     | -                            |
| `onApprove`     | `() =&gt; void`                 | `required`  | -                            |
| `onDeny`        | `() =&gt; void`                 | `required`  | -                            |
| `onAlwaysAllow` | `() =&gt; void`                 | `undefined` | -                            |
| `timeout`       | `number`                        | `undefined` | Auto-deny timeout in seconds |
