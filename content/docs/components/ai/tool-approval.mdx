## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/tool-approval.json
```

## Usage

```tsx
import { ToolApproval } from "@/components/ui/tool-approval";

export function Example() {
  return (
    <ToolApproval
      name="execute_shell"
      description="Run a shell command on the host machine"
      args={{ command: "rm -rf ./dist" }}
      risk="high"
      onApprove={() => console.log("approved")}
      onDeny={() => console.log("denied")}
      timeout={30}
    />
  );
}
```

## Props

| Prop            | Type                      | Default     | Description                                                          |
| --------------- | ------------------------- | ----------- | -------------------------------------------------------------------- |
| `name`          | `string`                  | `required`  | Name of the tool requesting approval.                                |
| `description`   | `string`                  | `undefined` | Short description of what the tool does.                             |
| `args`          | `Record<string, unknown>` | `undefined` | Key-value arguments the tool was invoked with.                       |
| `risk`          | `RiskLevel`               | `"low"`     | Risk level badge: `"low"`, `"medium"`, or `"high"`.                  |
| `onApprove`     | `() => void`              | `required`  | Called when the user presses `y` to approve.                         |
| `onDeny`        | `() => void`              | `required`  | Called when the user presses `n` to deny, or the timeout expires.    |
| `onAlwaysAllow` | `() => void`              | `undefined` | Called when the user presses `a` to always allow this tool.          |
| `timeout`       | `number`                  | `undefined` | Auto-deny countdown in seconds; denied automatically when it hits 0. |

## Examples

### Low-risk tool

```tsx
<ToolApproval
  name="read_file"
  description="Read a file from the workspace"
  args={{ path: "src/index.ts" }}
  risk="low"
  onApprove={() => {}}
  onDeny={() => {}}
/>
```

### High-risk with timeout

```tsx
<ToolApproval
  name="execute_shell"
  args={{ command: "DROP TABLE users;" }}
  risk="high"
  onApprove={() => {}}
  onDeny={() => {}}
  timeout={15}
/>
```

### With always-allow option

```tsx
<ToolApproval
  name="write_file"
  description="Write content to a file"
  args={{ path: "config.json", content: "{}" }}
  risk="medium"
  onApprove={() => {}}
  onDeny={() => {}}
  onAlwaysAllow={() => {}}
/>
```
