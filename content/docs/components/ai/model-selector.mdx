## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/model-selector.json
```

## Usage

```tsx
import { ModelSelector } from "@/components/ui/model-selector";

export function Example() {
  return (
    <ModelSelector
      models={[
        { id: "gpt-4o", name: "GPT-4o", provider: "OpenAI", context: 128000 },
        {
          id: "claude-3-5-sonnet",
          name: "Claude 3.5 Sonnet",
          provider: "Anthropic",
          context: 200000,
        },
      ]}
      selected="gpt-4o"
      onSelect={(id) => console.log("selected", id)}
    />
  );
}
```

## Props

| Prop              | Type                   | Default    | Description                                                                               |
| ----------------- | ---------------------- | ---------- | ----------------------------------------------------------------------------------------- |
| `models`          | `ModelOption[]`        | `required` | Array of models. Each has `id`, `name`, `provider`, and optional `context` (token count). |
| `selected`        | `string`               | `required` | The `id` of the currently selected model.                                                 |
| `onSelect`        | `(id: string) => void` | `required` | Called with the model `id` when the user presses Enter.                                   |
| `showContext`     | `boolean`              | `true`     | Display the context-window size next to each model.                                       |
| `showProvider`    | `boolean`              | `true`     | Display the provider name next to each model.                                             |
| `groupByProvider` | `boolean`              | `false`    | Group models under provider headings.                                                     |

## Examples

### Minimal list without context

```tsx
<ModelSelector
  models={[
    { id: "gpt-4o-mini", name: "GPT-4o Mini", provider: "OpenAI" },

  onSelect={(id) => console.log(id)}
  showContext={false}

```
