## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/chat-thread.json
```

## Usage

```tsx
import { ChatThread } from "@/components/ui/chat-thread";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <ChatThread {...demoProps} />;
}
```
