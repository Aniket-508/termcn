## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/path-input.json
```

## Usage

```tsx
import { PathInput } from "@/components/ui/path-input";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <PathInput {...demoProps} />;
}
```
