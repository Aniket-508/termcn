## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/masked-input.json
```

## Usage

```tsx
import { MaskedInput } from "@/components/ui/masked-input";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <MaskedInput {...demoProps} />;
}
```

## Props

| Prop          | Type                                                                                                        | Default     | Description |
| ------------- | ----------------------------------------------------------------------------------------------------------- | ----------- | ----------- |
| `mask`        | `string; // e.g. '(###) ###-####' value?: string; // raw digits only onChange?: (value: string) =&gt; void` | `required`  | -           |
| `onSubmit`    | `(value: string) =&gt; void`                                                                                | `undefined` | -           |
| `label`       | `string`                                                                                                    | `undefined` | -           |
| `placeholder` | `string`                                                                                                    | `undefined` | -           |
| `autoFocus`   | `boolean`                                                                                                   | `false`     | -           |
| `id`          | `string`                                                                                                    | `undefined` | -           |
| `width`       | `number`                                                                                                    | `40`        | -           |
