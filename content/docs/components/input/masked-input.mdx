## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/masked-input.json
```

## Usage

```tsx
import { MaskedInput } from "@/components/ui/masked-input";

export function Example() {
  return (
    <MaskedInput
      mask="(###) ###-####"
      label="Phone Number"
      placeholder="(555) 123-4567"
    />
  );
}
```

## Props

| Prop          | Type                      | Default      | Description                                          |
| ------------- | ------------------------- | ------------ | ---------------------------------------------------- |
| `mask`        | `string`                  | **required** | Format mask where `#` represents a digit placeholder |
| `value`       | `string`                  | `undefined`  | Controlled raw digit string (unformatted)            |
| `onChange`    | `(value: string) => void` | `undefined`  | Called with the raw digit string on change           |
| `onSubmit`    | `(value: string) => void` | `undefined`  | Called with the raw digit string on Enter            |
| `label`       | `string`                  | `undefined`  | Label text displayed above the input                 |
| `placeholder` | `string`                  | `undefined`  | Placeholder shown when the input is empty            |
| `autoFocus`   | `boolean`                 | `false`      | Whether to focus the input on mount                  |
| `id`          | `string`                  | `undefined`  | Unique identifier for focus management               |
| `width`       | `number`                  | `40`         | Width of the input box in columns                    |

## Examples

### Phone Number

```tsx
<MaskedInput mask="(###) ###-####" label="Phone" placeholder="(555) 123-4567" />
```

### Date Input

```tsx
<MaskedInput mask="##/##/####" label="Date of Birth" placeholder="MM/DD/YYYY" />
```

### Credit Card

```tsx
<MaskedInput mask="####-####-####-####" label="Card Number" width={30} />
```

### Controlled Value

```tsx
<MaskedInput
  mask="###-##-####"
  label="SSN"
  value={ssn}
  onChange={setSsn}
  onSubmit={(val) => verify(val)}
/>
```
