## Installation

```bash
npx shadcn@latest add https://termcn.vercel.app/r/password-input.json
```

## Usage

```tsx
import { PasswordInput } from "@/components/ui/password-input";

export function Example() {
  return (
    <PasswordInput
      label="Password"
      placeholder="Enter password"
      showToggle
      onSubmit={(value) => console.log("Password:", value)}
    />
  );
}
```

## Props

| Prop          | Type                                                                                         | Default     | Description                                      |
| ------------- | -------------------------------------------------------------------------------------------- | ----------- | ------------------------------------------------ |
| `value`       | `string`                                                                                     | `undefined` | Controlled input value                           |
| `onChange`    | `(value: string) => void`                                                                    | `undefined` | Called when the input value changes              |
| `onSubmit`    | `(value: string) => void`                                                                    | `undefined` | Called when Enter is pressed                     |
| `placeholder` | `string`                                                                                     | `""`        | Placeholder shown when the input is empty        |
| `mask`        | `string`                                                                                     | `"●"`       | Character used to mask each input character      |
| `showToggle`  | `boolean`                                                                                    | `false`     | Show a Ctrl+H hint to toggle password visibility |
| `label`       | `string`                                                                                     | `undefined` | Label text displayed above the input             |
| `id`          | `string`                                                                                     | `undefined` | Unique identifier for focus management           |
| `borderStyle` | `"single" \| "double" \| "round" \| "bold" \| "singleDouble" \| "doubleSingle" \| "classic"` | `"round"`   | Border style for the input box                   |
| `paddingX`    | `number`                                                                                     | `1`         | Horizontal padding inside the border             |
| `width`       | `number`                                                                                     | `undefined` | Width of the input box in columns                |
| `cursor`      | `string`                                                                                     | `"█"`       | Cursor character shown when focused              |

## Examples

### With Visibility Toggle

```tsx
<PasswordInput label="Password" showToggle placeholder="Enter password" />
```

### Custom Mask Character

```tsx
<PasswordInput label="PIN" mask="*" placeholder="Enter PIN" />
```

### Fixed Width

```tsx
<PasswordInput label="Secret Key" width={30} borderStyle="double" />
```
