
## Usage

```tsx
import { NumberInput } from "@/components/ui/number-input";

export function Example() {
  const demoProps = {
    // See the live preview for a working example.
  };

  return <NumberInput {...demoProps} />;
}
```

| Prop          | Type                                                                                            | Default     | Description                                                          |
| ------------- | ----------------------------------------------------------------------------------------------- | ----------- | -------------------------------------------------------------------- |
| `value`       | `number`                                                                                        | `undefined` | -                                                                    |
| `onChange`    | `(value: number) =&gt; void`                                                                    | `undefined` | -                                                                    |
| `onSubmit`    | `(value: number) =&gt; void`                                                                    | `undefined` | -                                                                    |
| `min`         | `number`                                                                                        | `undefined` | -                                                                    |
| `max`         | `number`                                                                                        | `undefined` | -                                                                    |
