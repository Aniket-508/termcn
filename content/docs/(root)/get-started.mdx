description: "Install terminal components, hooks, and themes from the registry."
---

## Install a Component

```bash
npx shadcn@latest add https://termcn.vercel.app/r/spinner.json
```

## Install a Theme

```bash
npx shadcn@latest add https://termcn.vercel.app/r/theme-dracula.json
```

## Wrap Your App

```tsx
import { ThemeProvider } from "@/components/ui/theme-provider";
import { draculaTheme } from "@/lib/terminal-themes/dracula";

export function App() {
  return (
    <ThemeProvider theme={draculaTheme}>{/* your Ink tree */}</ThemeProvider>
  );
}
```
